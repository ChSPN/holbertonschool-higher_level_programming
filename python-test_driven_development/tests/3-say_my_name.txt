#!/usr/bin/python3
"""
Ce module contient des tests pour la fonction say_my_name.
"""

import unittest
from 3-say_my_name import say_my_name

class TestSayMyName(unittest.TestCase):
    """
    Cette classe teste la fonction say_my_name.
    """

    def test_basic(self):
        """
        Teste si la fonction fonctionne avec un prénom et un nom de famille.
        """
        self.assertEqual(say_my_name("John", "Doe"), "My name is John Doe")

    def test_only_first_name(self):
        """
        Teste si la fonction fonctionne avec seulement un prénom.
        """
        self.assertEqual(say_my_name("John", ""), "My name is John ")

    def test_only_last_name(self):
        """
        Teste si la fonction fonctionne avec seulement un nom de famille.
        """
        self.assertEqual(say_my_name("", "Doe"), "My name is  Doe")

    def test_no_names(self):
        """
        Teste si la fonction fonctionne sans prénom ni nom de famille.
        """
        self.assertEqual(say_my_name("", ""), "My name is  ")

    def test_non_string_input(self):
        """
        Teste si la fonction lève une TypeError lorsqu'elle reçoit une entrée non valide.
        """
        with self.assertRaises(TypeError):
            say_my_name(123, 456)

if __name__ == '__main__':
    unittest.main()
