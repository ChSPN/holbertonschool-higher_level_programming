import unittest
from 3-say_my_name import say_my_name

class TestSayMyName(unittest.TestCase):
    def test_basic(self):
        self.assertEqual(say_my_name("John", "Doe"), "My name is John Doe")

    def test_only_first_name(self):
        self.assertEqual(say_my_name("John", ""), "My name is John ")

    def test_only_last_name(self):
        self.assertEqual(say_my_name("", "Doe"), "My name is  Doe")

    def test_no_names(self):
        self.assertEqual(say_my_name("", ""), "My name is  ")

    def test_non_string_input(self):
        with self.assertRaises(TypeError):
            say_my_name(123, 456)

if __name__ == '__main__':
    unittest.main()
